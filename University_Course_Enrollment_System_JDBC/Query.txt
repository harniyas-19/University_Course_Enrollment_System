CREATE TABLE Course_tbl (
    Course_ID        VARCHAR2(10) PRIMARY KEY,
    Course_title     VARCHAR2(150) NOT NULL,
    Credits          NUMBER(2) CHECK (CREDITS > 0),
    Capacity        NUMBER(4) CHECK (CAPACITY > 0),
    Enrolled_count   NUMBER(4) DEFAULT 0 CHECK (ENROLLED_COUNT >= 0)
);

CREATE TABLE Enrollment_tbl1 (
    Enrollment_ID    NUMBER(8) PRIMARY KEY,
    Course_ID        VARCHAR2(10),
    Student_ID       VARCHAR2(12) NOT NULL,
    Student_name     VARCHAR2(100) NOT NULL,
    Enrollment_date  DATE DEFAULT SYSDATE,
    Status           VARCHAR2(10) CHECK (STATUS IN ('ENROLLED','DROPPED')),

    CONSTRAINT FK_COURSE
    FOREIGN KEY (Course_ID)
    REFERENCES Course_tbl(Course_ID)
);
